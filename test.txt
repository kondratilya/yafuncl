baseFactory = inner {               # Фабрика, Возвращает функцию для перевода числа из системы счисления в параметре
    (base, ) = arguments;
    @?baseFactory;
    @base;
    @0 = outer 0;
    @mutable sum = 0;
    @mutable multiplyer = outer 1;
    {
        @?inside;
        @(digits = arguments);
        @mutable i = digits=>length;
        while {
            @? insideWhile;
            @+base;
            @i=--i;
            return 0 if i<0;
            sum = sum + digits[i] * multiplyer;
            @?here;
            multiplyer = multiplyer * base;                 # multiplyer никогда не равен "0"
        };
        @@sum;
    };
};
